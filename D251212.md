## 필수  문제 1에 필요한 개념
- 클래스 만들기 
            -생성자(객체 속성을 초기화하는 특별한 함수)
            -메서드(함수)
            -속성(클래스내에 있는 변수. aka 멤버변수)
'''dart
// class 클래스명 {이 안에서 구성요소 배치}
class Score {
       // 속성 : 점수(숫자)
       int point
       // 메서드 : 점수(속성중 하나)를 출력하는 기능
       //반환타입 함수이름(매개변수){실행할 코드}
       void showInfo(){
        print("점수: $point");
       }
       // 생성자 : 점수를 초기화
       Score(this.point);
}
'''
- 상속
    - 특정 클래스의 속성과 메서드들을 모두 포함하여
    확장해서 구현하고 싶을때
    - 'extends' 라는 키워드 사용!
''' dart
class StudentScore extends Score {
        // Score 클래스의 구성요소가 포함됨
        // 상속은 반드시 하나의 클래스만 가능
        // 현재 속성 : point, 메서드: showInfo 가지고 있음!
String name;
     //@override => 이노테이션 => 코드의 메타데이터 => 안붙여도 실행하는데 문제X
     //재정의 했다는걸 다른 개발자에게, 혹은 자신에게 보기 편하게 하려고 강력히 사용 권장!


@override
      void showInfo(){
          print("이름 : $name, 점수: $point");
  }
      //생성자
      StudentScore(this.name, super.point);


}
'''
